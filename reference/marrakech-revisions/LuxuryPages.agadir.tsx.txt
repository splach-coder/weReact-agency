function DetailPage({ locale, kind, slug }: { locale: string; kind: 'transfer' | 'fleet' | 'excursion'; slug: string }) {
  const item = kind === 'transfer' ? getRoute(slug) : kind === 'fleet' ? getFleetItem(slug) : getExcursion(slug);
  if (!item) return null;
  const title = 'title' in item ? item.title : 'name' in item ? item.name : '';
  const label = kind === 'transfer' ? 'Private transfer' : kind === 'fleet' ? item.category : 'Private excursion';
  const secondary = kind === 'transfer' ? item.distance : kind === 'fleet' ? item.capacity : item.duration;
  const back = kind === 'transfer' ? 'transfers' : kind === 'fleet' ? 'fleet' : 'excursions';
  return <div className="luxury-shell"><PageHero locale={locale} label={label} title={title} image={item.image} /><section className="luxury-section"><div className="luxury-container luxury-detail"><div className="luxury-detail__image"><Image src={item.image} alt={title} fill sizes="(max-width: 980px) 100vw, 60vw" /></div><div className="luxury-detail__copy"><span className="luxury-kicker">Just Luxury Transfers</span><h2>Travel well. Arrive ready.</h2><p>{item.description}</p><div className="luxury-detail__facts"><div className="luxury-fact"><strong>{secondary}</strong><span>{kind === 'transfer' ? 'Typical journey' : kind === 'fleet' ? 'Capacity' : 'Experience length'}</span></div><div className="luxury-fact"><strong>Private</strong><span>Service type</span></div></div><ul className="luxury-list"><li>Professional driver, carefully selected for your journey</li><li>Clean, comfortable vehicle with room for your plans</li><li>Direct WhatsApp support before and during your trip</li></ul><div className="luxury-hero__links"><Link className="luxury-button" href={`/${locale}/book`}>Book this journey <ArrowUpRight size={15} /></Link><Link className="luxury-button luxury-button--outline" href={`/${locale}/${back}`}><ArrowLeft size={15} /> Back to {back}</Link></div></div></div></section><section className="luxury-section luxury-section--tight"><div className="luxury-container"><div className="luxury-section-heading"><div><span className="luxury-kicker">Need something different?</span><h2>We can tailor the details.</h2></div><p>Share your pickup, timing, group size, and any special requests. We will recommend the right option.</p></div><Link className="luxury-button" href={`/${locale}/contact`}>Talk to our team <ArrowUpRight size={15} /></Link></div></section></div>;
}

export function TransferDetail({ locale, slug }: { locale: string; slug: string }) { return <DetailPage locale={locale} kind="transfer" slug={slug} />; }
export function FleetDetail({ locale, slug }: { locale: string; slug: string }) { return <DetailPage locale={locale} kind="fleet" slug={slug} />; }
export function ExcursionDetail({ locale, slug }: { locale: string; slug: string }) { return <DetailPage locale={locale} kind="excursion" slug={slug} />; }
